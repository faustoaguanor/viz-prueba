import nltk

from nltk.stem import PorterStemmer

stemmer = PorterStemmer()

word = "change"
print("Lematización:", lemmatizer.lemmatize(
    word, pos=wordnet.VERB))  # Lematización
print("Stemming:", stemmer.stem(word))  # Stemming

Remover las stop words (convirtiendo a minusculas para evitar problemas de mayusculas), eliminando números y signos de puntuación, y filtrando palabras con menos de 3 caracteres.

from nltk.corpus import stopwords
stop_words = set(stopwords.words('english'))
filtered_tokens = [word.lower() for word in tokens if word.isalpha() and word.lower() not in stop_words and len(word) > 2]
print(filtered_tokens[:20]) # Imprime los primeros 20 tokens filtrados
['seventh', 'sally', 'trurl', 'perfection', 'led', 'good', 'lem', 'translated', 'michael', 'kandel', 'universe', 'infinite', 'bounded', 'therefore', 'beam', 'light', 'whatever', 'direction', 'may', 'travel']

Usar Bag of Words para contar la frecuencia de las palabras

bow = {}
for word in filtered_tokens:
    if word in bow:
        bow[word] += 1
    else:
        bow[word] = 1
# Convertir el diccionario a un DataFrame para visualización
bow_df = pd.DataFrame(list(bow.items()), columns=["word", "frequency"])
# Visualizar las palabras más frecuentes usando Altair
chart = (
    alt.Chart(bow_df.sort_values(by="frequency", ascending=False).head(20))
    .mark_bar()
    .encode(x=alt.X("word", sort="-y"), y="frequency")
    .properties(title="Top 20 Palabras Más Frecuentes")
)

from wordcloud import WordCloud

wordcloud = WordCloud(width=800, height=400, background_color='white').generate_from_frequencies(bow)

Word cloud and bar chart side by side

words = []
types = []

# Itera sobre cada entrada de la lista POS
for p in pos_tags:
    word = p[0].lower()  # obtener la palabra en minúsculas
    tag = p[1]  # obtener la etiqueta gramatical

    # Usamos 'stop_words' y aplicamos las reglas de limpieza previas (longitud > 2)
    if word.isalpha() and word not in stop_words and len(word) > 2:
        words.append(word)  # añadir palabra limpia a la lista
        # clasificar el tipo según el prefijo de la etiqueta
        if tag.startswith("N"):
            types.append("Noun")
        elif tag.startswith("V"):
            types.append("Verb")
        elif tag.startswith("J"):
            types.append("Adjective")
        elif tag.startswith("R"):
            types.append("Adverb")
        else:
            types.append("Other")

# Crear DataFrame
df = pd.DataFrame({"word": words, "type": types})

# Base de los dos subgráficos
base = alt.Chart(df)

# Selección por filtro (al hacer clic en una barra del primer gráfico)
selection = alt.selection_point(fields=["type"])

# Condición de opacidad: resaltar si está seleccionado
opacity_condition = alt.condition(selection, alt.value(1), alt.value(0.1))

# Gráfico 1: barras de frecuencia por tipo gramatical
chart1 = (
    base.mark_bar()
    .encode(
        alt.X("type", sort="-y"),  # ordenar de mayor a menor frecuencia
        alt.Y("count()"),  # agregación correcta en Altair
        opacity=opacity_condition,
    )
    .add_params(selection)
)

# Gráfico 2: barras de palabras, filtradas por la selección del gráfico 1
chart2 = (
    base.mark_bar()
    .encode(
        # ordenar palabras por frecuencia
        alt.X("word", sort="-y"), alt.Y("count()")
    )
    .transform_filter(selection)
    .transform_window(
        rank="rank(count())", sort=[alt.SortField("count()", order="descending")]
    )
    .transform_filter(alt.datum.rank <= 20)
)
